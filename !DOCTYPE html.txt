<script>
  const input = document.getElementById('searchInput');
  const typingLabel = document.getElementById('typingLabel');
  const searchBtn = document.getElementById('searchBtn');
  const results = document.getElementById('results');
  const photosGrid = document.getElementById('photosGrid');
  const closeBtn = document.getElementById('closeBtn');

  const text = "the real jordan belfort";
  let i = 0;
  let started = false;

  function startSequence() {
    if (started) return;
    started = true;
    setTimeout(() => { typingLabel.classList.add('show'); }, 600);
    setTimeout(() => { typeChar(); }, 1000);
  }

  function typeChar() {
    if (i < text.length) {
      input.value += text[i];
      i++;
      setTimeout(typeChar, 75 + Math.random() * 55);
    } else {
      setTimeout(() => {
        searchBtn.style.transform = 'scale(0.9)';
        setTimeout(() => {
          searchBtn.style.transform = 'scale(1)';
          showResults();
        }, 150);
      }, 700);
    }
  }

  function showResults() {
    results.classList.add('visible');
    const cards = photosGrid.querySelectorAll('.photo-card');
    cards.forEach((card, idx) => {
      setTimeout(() => { card.classList.add('show'); }, idx * 120);
    });
  }

  closeBtn.addEventListener('click', () => {
    results.classList.remove('visible');
    setTimeout(() => {
      input.value = '';
      i = 0;
      started = false;
      typingLabel.classList.remove('show');
      photosGrid.querySelectorAll('.photo-card').forEach(c => c.classList.remove('show'));
      setTimeout(startSequence, 1500);
    }, 600);
  });

  setTimeout(startSequence, 1200);
</script>
